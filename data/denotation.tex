\begin{denotation}[3cm]
\item[$ T $]  扩散过程总时间步数
\item[$ \beta_t $]  第 $ t $ 步的噪声调度参数
\item[$ x_0 $]  原始数据样本
\item[$ x_t $]  第 $ t $ 步加噪后的隐变量
\item[$ \epsilon_\theta $]  参数为 $ \theta $ 的去噪神经网络
\item[$ \mu_\theta $]  逆向过程的均值预测
\item[$ \sigma_t $]  逆向过程的标准差
\item[$ w $]  CFG（Classifier-Free Guidance）权重

\end{denotation}
